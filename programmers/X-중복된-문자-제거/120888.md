중복 제거 패턴

```js
function solution(my_string) {
  return [...new Set(my_string)].join("");
}
```

모범답안

```js
/**
 * @param {string} myString
 * @returns {string}
 */
function solution(myString) {
  const memo = {};
  return myString
    .split("")
    .filter((char) => {
      if (!memo[char]) {
        memo[char] = 1;
        return true;
      }
      return false;
    })
    .join("");
}
```
