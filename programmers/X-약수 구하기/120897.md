```js
function solution(n) {
  return Array(n)
    .fill(0)
    .map((v, index) => v + index + 1)
    .filter((v) => n % v === 0);
}
```

```js
/**
 * @param {number} n
 * @returns {array}
 */
function solution(n) {
  const answer = [];
  let num = 1;
  while (num !== n + 1) {
    if (n % num === 0) answer.push(num);
    num += 1;
  }
  return answer;
}
```
